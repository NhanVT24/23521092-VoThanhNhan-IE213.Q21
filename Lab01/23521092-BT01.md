//2.2 Thêm các document vào collection tên là employees (dùng insertMany):

db.employees.insertMany([
  { _id: 1, name: { first: "John", last: "Doe" }, age: 48 },
  { _id: 2, name: { first: "Jane", last: "Doe" }, age: 16 },
  { _id: 3, name: { first: "Alice", last: "A" }, age: 32 },
  { _id: 4, name: { first: "Bob", last: "B" }, age: 64 }
])

//2.3 Biến các biến trường id thành duy nhất:

db.employees.createIndex({ _id: 1 }, { unique: true }) 
//(do em dùng _id làm khóa chính nên không cần tạo index riêng cho trường id)

//2.4 tìm document có first name là John và last name là Doe:

db.employees.find({ "name.first": "John", "name.last": "Doe" })

//2.5 tìm những document có age lớn hơn 30 và nhỏ hơn 60:

db.employees.find({
  $and: [
    { age: { $gt: 30 } },
    { age: { $lt: 60 } }
  ]
})

//2.6 Thêm document mới vào collection

db.employees.insertMany([
  {
    id: 5,
    name: { first: "Rooney", middle: "K", last: "A" },
    age: 30
  },
  {
    id: 6,
    name: { first: "Ronaldo", middle: "T", last: "B" },
    age: 60
  }
])

// tìm document có middle 

db.employees.find({"name.middle" : {$exists: true}})

//2.7 Xóa trường middle name khỏi tất cả document:

db.employees.updateMany(
  { "name.middle": { $exists: true } },
  { $unset: { "name.middle": "" } }
)

//2.8 Thêm trường organization với giá trị là "UIT" vào tất cả document:

db.employees.updateMany(
  {},
  { $set: { organization: "UIT" } }
)

//2.9 Cập nhật trường organization thành "USSH" cho những document có id là 5 và 6:

db.employees.updateMany(
  { id: { $in: [5, 6] } },
  { $set: { organization: "USSH" } }
)

//2.10.1 Tính tổng và trung bình tuổi của nhân viên 2 organization là UIT và USSH:

db.employees.aggregate([
  { $match: { organization: { $in: ["UIT", "USSH"] } } },
  {
    $group: {
      _id: "$organization",
      totalAge: { $sum: "$age" },
      avgAge: { $avg: "$age" }
    }
  }
])
